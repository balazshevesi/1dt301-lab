cmake_minimum_required(VERSION 3.13)
include(./pico-sdk/external/pico_sdk_import.cmake)
project(pico_asm_blink C CXX ASM)

set(PICO_BOARD pico_w)        # change to "pico" if you have the non-W board
pico_sdk_init()


add_executable(main main.c link_shim.c)
target_link_libraries(main pico_stdlib hardware_gpio)

pico_enable_stdio_usb(main 1)
pico_enable_stdio_uart(main 0)

pico_add_extra_outputs(main) # creates .uf2 for easy flashing
